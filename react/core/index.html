function createDOM(fiber) {
  const dom =
    fiber.type == 'TEXT-ELEMENT'
      ? document.createTextNode('')
      : document.createElement(fiber.type)
  Object.keys(fiber.props)
    .filter((key) => key !== 'children')
    .forEach((key) => {
      dom[key] = fiber.props[key]
    })
  return dom
}
